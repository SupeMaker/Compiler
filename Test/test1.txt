#include<stdio.h>
#include "tress.h"
int main(void) {
    int a = 50;
    int sum = -1;

    // loop
    for (int i = 0; i < a * 2; i++) {
        // inner loop
        for (int j = 0; j < a ; j++) {
            if (j < a / 2 && j % 2 != 0 ) {
                sum = sum +  i * j / 2 + i * j;
            } else {
                while (sum > j || ! sum % 2 == 1) {
                    sum = sum - 3 / 2;
                }
            }
        }
    }
    cout << a;
    // print the value of `sum`

    return 0;
}
